> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Couvrir toutes les routes API du projet avec `bun test` + `mongodb-memory-server`, sans démarrer de serveur HTTP.

**Architecture:** Chaque fichier de test importe son handler directement après avoir mocké ses dépendances via `mock.module()` + `await import()` top-level. **Tech Stack:** `bun test` (intégré), `mongodb-memory-server` (devDependency), `bun:test` mock API

import { MongoClient } from "mongodb";
import type { Db } from "mongodb";

export async function startMemoryDb(): Promise<{ db: Db; stop: () => Promise<void> }> {
    const server = await MongoMemoryServer.create();
    const client = new MongoClient(server.getUri());
    await client.connect();
    const db = client.db("test");

        stop: async () => {
            await client.close();
            await server.stop();
        },
    };

import { beforeAll, afterAll, beforeEach, describe, test, expect, mock } from "bun:test";
import type { Db } from "mongodb";
import path from "path";

let db: Db;

// Filesystem virtuel : path → liste d'entrées (string[]) ou inexistant
const vfs: Record<string, string[]> = {};

function normPath(p: string): string {
    return p.replace(/\\/g, "/");
}

mock.module("fs/promises", () => ({
    readdir: async (dirPath: string) => {
        const key = normPath(dirPath);
        if (!(key in vfs)) throw Object.assign(new Error(`ENOENT: ${dirPath}`), { code: "ENOENT" });
        return vfs[key];
    },
    stat: async (filePath: string) => {
        const key = normPath(filePath);
        // C'est un dossier si la clé existe dans vfs, un fichier sinon
        return { isDirectory: () => key in vfs };
    },
}));

mock.module("@/lib/mongodb", () => ({ connectToDB: async () => db }));
mock.module("@/lib/withAdmin", () => ({ withAdmin: (fn: Function) => fn }));

const { GET } = await import("@/app/api/admin/sync/route");

const COURS_DIR = normPath(path.join(process.cwd(), "src", "cours"));

let stopDb: () => Promise<void>;
beforeAll(async () => {
    const { startMemoryDb } = await import("../../helpers/db");
    ({ db, stop: stopDb } = await startMemoryDb());
});
afterAll(() => stopDb());
beforeEach(async () => {
    await db.dropDatabase();
    // Réinitialise le filesystem virtuel
    for (const key of Object.keys(vfs)) delete vfs[key];
    vfs[COURS_DIR] = []; // dossier cours vide par défaut
});

describe("GET /api/admin/sync", () => {
    test("filesystem vide + MongoDB vide → diff vide", async () => {
        const res = await GET(new Request("http://localhost/api/admin/sync"), {});
        const body = await res.json();
        expect(res.status).toBe(200);
        expect(body.missingModules).toHaveLength(0);
        expect(body.missingSections).toHaveLength(0);
    });

    test("module présent dans le filesystem mais absent de MongoDB → missingModules", async () => {
        vfs[COURS_DIR] = ["javascript"];
        vfs[`${COURS_DIR}/javascript`] = []; // dossier vide

        const res = await GET(new Request("http://localhost/api/admin/sync"), {});
        const body = await res.json();
        expect(body.missingModules).toHaveLength(1);
        expect(body.missingModules[0].slug).toBe("javascript");
        expect(body.missingSections).toHaveLength(0);
    });

    test("module dans MongoDB + section manquante → missingSections avec prefill", async () => {
        // Module dans MongoDB
        const { insertedId } = await db.collection("modules").insertOne({
            title: "JavaScript",
            path: "javascript",
            iconName: "Code",
            sections: [],
            coefficients: [],
            associatedSae: [],
            instructors: [],
        });

        // Filesystem : module présent avec une section
        vfs[COURS_DIR] = ["javascript"];
        vfs[`${COURS_DIR}/javascript`] = ["1-le-dom"];
        vfs[`${COURS_DIR}/javascript/1-le-dom`] = ["Cours.tsx", "TP.tsx"];

        const res = await GET(new Request("http://localhost/api/admin/sync"), {});
        const body = await res.json();
        expect(body.missingModules).toHaveLength(0);
        expect(body.missingSections).toHaveLength(1);

        const item = body.missingSections[0];
        expect(item.sectionSlug).toBe("1-le-dom");
        expect(item.prefill.title).toBe("Le Dom");
        expect(item.prefill.order).toBe(1);
        expect(item.prefill.contents).toContain("cours");
        expect(item.prefill.contents).toContain("TP");
        expect(item.prefill.isAvailable).toBe(false);
    });

    test("section déjà dans MongoDB → non incluse dans missingSections", async () => {
        await db.collection("modules").insertOne({
            title: "JavaScript",
            path: "javascript",
            iconName: "Code",
            sections: [{ path: "1-le-dom", title: "Le Dom", order: 1, contents: ["cours"] }],
            coefficients: [],
            associatedSae: [],
            instructors: [],
        });

        vfs[COURS_DIR] = ["javascript"];
        vfs[`${COURS_DIR}/javascript`] = ["1-le-dom"];
        vfs[`${COURS_DIR}/javascript/1-le-dom`] = ["Cours.tsx"];

        const res = await GET(new Request("http://localhost/api/admin/sync"), {});
        const body = await res.json();
        expect(body.missingSections).toHaveLength(0);
    });
});
